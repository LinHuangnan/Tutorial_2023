  class Solution
  {
  	public:
  		static	ListNode* reverseList(ListNode* head)
  		{
              //总共定义三个指针来辅助反转
  			ListNode* pre = NULL; 
  			ListNode* cur = head;
  			ListNode* nex = NULL;
  			while(cur)
  			{
  				nex = cur->next; 
  				cur->next = pre;
  				pre = cur;
  				cur = nex;
